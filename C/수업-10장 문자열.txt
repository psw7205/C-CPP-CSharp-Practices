실습 5
#include <stdio.h>
#pragma warning(disable : 4996)

int main()
{
	char str[3][20] = {"Time is gold", "No pain no gain", "No sweat no sweet"};
	int i, j, cnt[3] = { 0 };

	for (i = 0; i < 3; i++)
	{
		printf("%s\n", str[i]);
	}

	for (i = 0; i < 3; i++)
	{
		for (j = 0; str[i][j]; j++)
		{
			if (str[i][j] == 'a')
			{
				cnt[i]++;
			}
		}
	}

	for(i = 0; i < 3; i++)
	{
		printf("%d\n", cnt[i]);
	}
	return 0;
}
