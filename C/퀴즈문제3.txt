#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

typedef struct box {
	char name[11];
	int price;
	int num;
}box;

int main()
{
	box ar[10];
	int i = 0, cnt = 0, max = 0, max2 = 0;
	char tmp[11];

	while (cnt < 10)
	{
		scanf("%s", tmp);
		if (strlen(tmp) == 1)
			break;
		strcpy(ar[cnt].name, tmp);
		scanf("%d %d", &ar[cnt].price, &ar[cnt].num);
		cnt++;
	}

	for (i = 0; i < cnt; i++)
	{
		if (max < ar[i].price)
			max = ar[i].price;

		if (max2 < ar[i].price * ar[i].num)
			max2 = ar[i].price * ar[i].num;
	}

	for (i = 0; i < cnt; i++)
	{
		if (max == ar[i].price)
		{
			printf("%s\n", ar[i].name);
			break;
		}
	}

	for (i = 0; i < cnt; i++)
	{
		if (max2 == ar[i].price * ar[i].num)
		{
			printf("%s\n", ar[i].name);
			break;
		}
	}

	return 0;
}

#include <stdio.h>
#pragma warning (disable:4996)

typedef struct st {
	char name[21];
	int kor;
	int math;
	double ave;
}st;

void insert(st *ar, int n)
{
	int i;
	for (i = 0; i < n; i++)
		scanf("%s %d %d", ar[i].name, &ar[i].kor, &ar[i].math);

}

st maxReturn(st *ar, int n)
{
	st max;
	int i;

	max.ave = 0.0;

	for (i = 0; i < n; i++)
		ar[i].ave = ar[i].math + ar[i].kor;

	for (i = 0; i < n; i++)
		ar[i].ave /= 2.0;

	for (i = 0; i < n; i++)
	{
		if (max.ave < ar[i].ave)
			max = ar[i];
	}

	return max;
}

int main()
{
	st ar[100];
	st tmp;
	int n;

	scanf("%d", &n);

	insert(ar, n);

	tmp = maxReturn(ar, n);

	printf("%s %d %d %.1lf", tmp.name, tmp.kor, tmp.math, tmp.ave);

	return 0;
}

#include <stdio.h>
#include <string.h>
#include <stdlib.h>
#pragma warning (disable:4996)

typedef struct stu {
	char *name;
	int year;
}stu;

int main()
{
	stu *ar;
	char name[101];
	int n, i;

	scanf("%d", &n);

	ar = (stu*)malloc(sizeof(stu)*n);

	for (i = 0; i < n; i++)
	{
		scanf("%s", name);
		ar[i].name = (char*)malloc(sizeof(char)*(strlen(name) + 1));
		strcpy(ar[i].name, name);
		scanf("%d", &ar[i].year);
		ar[i].year /= 10000;
		ar[i].year = 2018 - ar[i].year + 1;
	}

	for (i = 0; i < n; i++)
		printf("%s %d\n", ar[i].name, ar[i].year);

	for (i = 0; i < n; i++)
		free(ar[i].name);
	free(ar);

	return 0;
}
