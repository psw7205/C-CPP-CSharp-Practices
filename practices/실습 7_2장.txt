#include<stdio.h>
#pragma warning(disable:4996)

int main()
{
	int n, i, cnt = 0;
	int x[10];

	scanf("%d", &n);

	for (i = 0; i < 10; i++)
	{
		if (n == 0)
			break;
		x[i] = n % 10;
		n /= 10;
		cnt++;
	}

	for (i = 0; i < cnt; i++)
		printf(" %d", x[i]);

	return 0;
}

{

	int i, j, flag = 0;
	for (i = 0; i < 10; i++)
		scanf("%c", &x[i]);
	for (i = 0; i < 10; i++)
	{
		for (j = 0; j < 10; j++)
		{
			if (i == j)
				flag = 0;
			else
				flag = 1;

			if (x[i] == x[j] && flag == 1)
				break;
		}
		if (j == 10)
			printf(" %c", x[i]);
	}
	return 0;
}
