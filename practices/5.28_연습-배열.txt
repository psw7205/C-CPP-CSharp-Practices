#include<stdio.h>

int main()
{
	int x[2][3] = { {10,20,50},{20,30,40} };
	int i, j;
	
	for (i = 0; i < 2; i++)
	{
		for (j = 0; j < 3; j++)
		{
			printf("x[%d][%d] = %d\t", i, j, x[i][j]);
		}
		printf("\n");
	}

	return 0;
}

#include<stdio.h>
#pragma warning(disable:4996)

int main()
{
	int x[3][2] = { 0 };
	int i, j;
	double avg[2] = { 0.0 };
	
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 2; j++)
		{
			scanf("%d", &x[i][j]);
		}
	}

	for (i = 0; i < 3; i++)
	{
		avg[0] += x[i][0];
		avg[1] += x[i][1];
	}

	printf("%.1lf %.1lf", avg[0] / 3, avg[1] / 3);

	return 0;
}

{
	int x[3][3] = { 0 };
	int i, j, a, b, n;
	
	scanf("%d %d %d", &a, &b, &n);
	
	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
		{
			printf(" %d", x[i][j]);
		}
		printf("\n");

	}

	x[a-1][b-1] = n;

	for (i = 0; i < 3; i++)
	{
		for (j = 0; j < 3; j++)
		{
			printf(" %d", x[i][j]);
		}
		printf("\n");
	}

	return 0;
}
