{

#include <stdio.h>
#pragma warning(disable:4996)

int max1, max2;

void update_max(int x)
{
	if (max2 <= x && max1 >= x)
		max2 = x;

	if (max1 <= x)
	{
		max2 = max1;
		max1 = x;
	}

}

int main()
{
	int n, tmp;

	scanf("%d", &n);
	max1 = n;
	scanf("%d", &n);
	max2 = n;
	
	if (max1 < max2)
	{
		tmp = max1;
		max1 = max2;
		max2 = tmp;
	}
	
	while (1)
	{
		scanf("%d", &n);

		if (n == 0)
			break;

		update_max(n);
	}

	printf("%d %d", max1, max2);

	return 0;
}

int a, b, c;

int count_strike(int in_a, int in_b, int in_c)
{
	int cnt = 0;

	if (in_a == a)
		cnt++;

	if (in_b == b)
		cnt++;

	if (in_c == c)
		cnt++;
	
	return cnt;
}

int count_ball(int in_a, int in_b, int in_c)
{
	int cnt = 0;

	if (a == in_b || a == in_c)
		cnt++;
	if (b == in_a || b == in_c)
		cnt++;
	if (c == in_b || c == in_a)
		cnt++;

	return cnt;
}

int main()
{
	int a1, b1, c1;

	scanf("%d %d %d", &a, &b, &c);

	while(1)
	{
		scanf("%d %d %d", &a1, &b1, &c1);
		printf("%dS%dB\n", count_strike(a1, b1, c1), count_ball(a1, b1, c1));
		if (count_strike(a1, b1, c1) == 3)
			break;
	}

	return 0;

#pragma warning(disable:4996)

void print_line(int x, int y, int z)
{
	int i, j, k = 0;

	for (i = 0; i < x + y; i++)
	{
		for (j = 0; j < x-i; j++)
			printf(" ");

		for (j = 0; j < y+k; j++)
			printf("*");

		for (j = 0; j < z-i; j++)
			printf(" ");
		k += 2;
		printf("\n");
	}
}

int main()
{
	int n;

	scanf("%d", &n);

	print_line(n - 1, 1, n - 1);

	return 0;
}

{
	int i, cnt = 0, tmp;

	{
		tmp = i;
		while (tmp)
		{
			if ((tmp % 10) == 3)
				cnt++;
			tmp /= 10;
		}
	}

	return cnt;
}

{
	int n;

	scanf("%d", &n);

	printf("%d", count_3(n));

#include <stdio.h>
#pragma warning(disable:4996)

int is_prime(int x)
{
	int i;

	for (i = 2; i <= x; i++)
	{
		if (x%i == 0)
			break;
	}

	if (i == x)
		return 1;
	else
		return 0;
}

int next_prime(int x)
{
	int tmp = x;

	while (1)
	{
		tmp++;
		if (is_prime(tmp))
			break;
	}

	return tmp;
}

int main()
{
	int n, m, i, j;

	scanf("%d %d", &n, &m);

	for (i = 0; i < m; i++)
	{
		printf(" %d", next_prime(n));
		n = next_prime(n);
	}

	return 0;
}
