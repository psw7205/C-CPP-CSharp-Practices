#include <stdio.h>
#pragma warning (disable:4996)

int main()
{
	int n, m, tmp, len;
	int *p, *q;
	int arr[100];

	scanf("%d", &n);
	len = n;

	for(p = arr; p < arr + n; p++)
	{
		scanf("%d", p);
	}

	while (1)
	{
		scanf("%d %d", &n, &m);

		if(n == -1 || m == -1)
			break;

		if(n > m)
		{
			tmp = n;
			n = m;
			m = tmp;
		}


		for(p = arr + n, q = arr + m; p <q; p++, q--)
		{
			tmp = *q;
			*q = *p;
			*p = tmp;
		}

		for(p = arr; p < arr + len; p++)
		{
			printf(" %d", *p);
		}
		printf("\n");

	}


	return 0;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

struct student{
	char name[21];
	char a;
	double result[4];
	double ave;
	int pass;
};

void putInfo(struct student *p);
struct student getMaxInfo(struct student *p, int N);

int main()
{
	int i, N;

	struct student st[30], temp;
	scanf("%d", &N);
	
	for(i = 0; i < N; i++){
		putInfo(&st[i]);
	}

	temp = getMaxInfo(st, N);
	printf("%s %.1lf", temp.name, temp.ave);

	return 0;
}

void putInfo(struct student *p){
	
	int i;

	scanf("%s %c %lf %lf %lf %lf", p->name, &p->a, &p->result[0],  &p->result[1],  &p->result[2],  &p->result[3]);

	if(p->result[0] >= 60  && p->result[1] >= 60 &&p->result[2]>= 60  &&p->result[3] >= 60)
		p->pass = 1;
	else
		p->pass = 0;

	if(p->a == 'L')
	{
		p->result[0] *= 1.2;
		p->result[1] *= 1.2;
	}

	if(p->a == 'S')
	{
		p->result[2] *= 1.2;
		p->result[3] *= 1.2;
	}

	p->ave = 0;

	for(i = 0 ; i < 4; i++)
	{
		p->ave += p->result[i];
	}

	p->ave /= 4;
	
}

struct student getMaxInfo(struct student *p, int N)
{
	int i, j, cnt = 0;
	double max, max2;
	struct student temp;

	for(i = 0; i < N; i++)
	{
		if(p[i].pass == 1)
		{
			max = p[i].ave;
			max2 = p[i].result[2] + p[i].result[3];
			break;
		}
	}

	for(i = 0; i < N; i++)
	{
		if(p[i].pass == 1 && max <= p[i].ave && max2 <  p[i].result[2] + p[i].result[3])
		{
			max = p[i].ave;
			max2 = p[i].result[2] + p[i].result[3];
		}
	}

	for(i = 0; i < N; i++)
	{
		if(max == p[i].ave && max2 == p[i].result[2] + p[i].result[3])
		{	
			temp = p[i];
		}
	}

	return temp;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

struct person {
	char name[11];
	double a;
	double b;
	double result;
	int year;
	int month;
	int day;
};

void swap(struct person *p, struct person *q);

int main()
{
	struct person ar[100];
	int n, m;
	int i, j;


	scanf("%d %d", &n, &m);

	for(i = 0; i < n; i++)
	{
		scanf("%s %d-%d-%d %lf %lf", ar[i].name, &ar[i].year, &ar[i].month,  &ar[i].day, &ar[i].a, &ar[i].b);
		ar[i].result = ar[i].a * 0.8 + ar[i].b * 0.2;
	}
	for(i = 0; i < n - 1; i++)
	{
		for(j = 0; j < n -1 - i; j++)
		{
			if(ar[j].year < ar[j + 1].year)
				swap(&ar[j], &ar[j+1]);
			else if (ar[j].year== ar[j + 1].year && ar[j].month < ar[j + 1].month)
					swap(&ar[j], &ar[j+1]);
			else if( ar[j].year== ar[j + 1].year && ar[j].month == ar[j + 1].month && ar[j].day < ar[j + 1].day)
				swap(&ar[j], &ar[j+1]);
		}
	}

	for(i = 0; i < n - 1; i++)
	{
		for(j = 0; j < n -1 - i; j++)
		{
			if(ar[j].a < ar[j + 1].a)
				swap(&ar[j], &ar[j+1]);
		}
	}

	for(i = 0; i < n - 1; i++)
	{
		for(j = 0; j < n -1 - i; j++)
		{
			if(ar[j].result < ar[j + 1].result)
				swap(&ar[j], &ar[j+1]);
		}
	}

	for(i = 0; i < m; i++)
	{
		printf("%s %.1lf\n", ar[i].name, ar[i].result);
	}

	return 0;
}

void swap(struct person *p, struct person *q)
{
	struct person temp;

		temp = *p;
		*p = *q;
		*q = temp;
}
