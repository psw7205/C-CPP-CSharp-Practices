문제 1
#include <stdio.h>
#pragma warning(disable:4996)

int consonant(char c)
{
	char a[10] = { 'a','e','i','o','i','A','E','O','I','U'};
	int i;

	for(i = 0; i < 10; i++)
	{
		if (a[i] == c)
			return 0;
	}

	return 1;
}

int main()
{
	int n, i, j, max, min;
	char c[31] = { '\0' };
	int cnt[31] = { 0 };

	scanf("%d", &n);
	getchar();

	for (i = 0; i < n; i++)
		scanf("%c", &c[i]);

	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			if (c[i] == c[j])
				cnt[i]++;
		}
	}

	max = min = cnt[0];
	
	for (i = 1; i < n; i++)
	{
		if (max < cnt[i] && consonant(c[i]))
			max = cnt[i];

		if (min > cnt[i] && consonant(c[i]))
			min = cnt[i];
	}

	for (i = 0; i < n; i++)
	{
		if (!consonant(c[i]) && cnt[i] == 1)
		{
			min = cnt[i];
			break;
		}
	}

	printf("%d %d", min, max);

	return 0;
}

int consonant(char c)
{

	int i;

	for(i = 0; i < 5; i++)
	{
		if (a[i] == c)
			return 0;
	}

	return 1;
}

int main()
{
	int i;
	char c[10] = { '\0' };
	int cnt[10] = { 0 };

	for (i = 0; i < 10; i++)
		scanf("%c", &c[i]);

	for (i = 0; i < 10; i++)
	{
		if (consonant(c[i]))
			cnt[i]++;
	}

	for (i = 0; i < 8; i++)
	{
		if (cnt[i] == 1 && cnt[i + 1] == 0 && cnt[i + 2] == 1)
			printf("%c%c%c\n", c[i], c[i + 1], c[i + 2]);
	}

	return 0;
}

문제 E
#include <stdio.h>
#pragma warning(disable:4996)

int main()
{
	int n, i = 0, j, k, tmp, max = 0, min = 0;
	int a[10] = { 0 };

	scanf("%d", &n);

	while (n)
	{
		a[i] = n % 10;
		i++;
		n /= 10;
	}

	for (j = 0; j < i - 1; j++)
	{
		for (k = 0; k < i - 1 - j; k++)
		{
			if (a[k] < a[k + 1])
			{
				tmp = a[k];
				a[k] = a[k + 1];
				a[k + 1] = tmp;
			}
		}
	}

	for (j = 0; j < 3; j++)
	{
		max = max * 10 + a[j];
		min = min * 10 + a[i - j - 1];
	}

	printf("%d\n%d", min, max);

	return 0;
}

			printf("draw\n");
		else if (a > ans)
			printf("B\n");
		else if (b > ans)
			printf("A\n");
		else if (a > b)
			printf("A\n");
		else if (a < b)
			printf("B\n");
		else
			printf("draw\n");
	}

	return 0;
}

문제 G
#include<stdio.h>
#pragma warning(disable:4996)

int main()
{
	int l = 0, s = 0, sum = 0, etc = 0;
	char c;

	while (1)
	{
		scanf("%c", &c);

		if (c == '*')
			break;

		if (c >= 'A' && c <= 'Z')
			l++;
		else if (c >= 'a' && c <= 'z')
			s++;
		else if (c >= '0' && c <= '9')
			sum += c - '0';
		else
			etc++;
	}

	printf("%d %d %d %d", l, s, sum, etc);

	return 0;
}
