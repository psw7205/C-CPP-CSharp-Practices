#include <stdio.h>
#include <stdlib.h>	
#pragma warning (disable:4996)

int main()
{
	int n, i;
	int *score = NULL;
	double sum = 0.0;

	scanf("%d", &n);

	score = (int*)malloc(sizeof(int)*n);

	for (i = 0; i < n; i++)
	{
		scanf("%d", &score[i]);
		sum += score[i];
	}

	for (i = 0; i < n; i++)
		printf("%d", score[i]);

	printf("\n%lf", sum /= n);

	free(score);

	return 0;
}

int main()
{

	int **p = NULL;

	scanf("%d", &row);
	scanf("%d", &col);

	p = (int**)malloc(sizeof(int*)*row);
	
	for(i = 0; i < row; i++)
		p[i] = (int*)malloc(sizeof(int)*col);
	
	for (i = 0; i < row; i++)
		free(p[i]);
	
	free(p);

	return 0;
}
