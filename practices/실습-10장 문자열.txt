#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

int main()
{
	char* num[9] = { "one ", "two ", "three ", "four ", "five ", "six ", "seven ", "eight ", "nine " };
	char str[10000] = { '\0' };

	int a, cnt = 0, i = 0; 
	int tmp[5] = { 0 };

	scanf("%d", &a);

	while (a > 0)
	{
		tmp[cnt] = a % 10;
		a /= 10;
		cnt++;
	}
	
	for (i = 0; i < cnt; i++)
	{
		if (tmp[cnt - i - 1] != 0)
		{
			strcat(str, num[tmp[cnt - i - 1] - 1]);

			switch (cnt - i)
			{
			case 4:
				strcat(str, "THO ");
				break;
			case 3:
				strcat(str, "HUN ");
				break;
			case 2:
				strcat(str, "TEN ");
				break;
			default:
				break;
			}
		}
	}

	printf("%s", str);

	return 0;
}

int check(char* str)
{
	char* p = &str[0];
	char* q = &str[strlen(str) - 1];
	int cnt = 0;

	while (p <= q)
	{
		p++;
		q--;
		if (*p != *q)
			return 0;
	}

	return 1;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

int main()
{
	char str1[81] = { '\0' };
	char str2[11] = { '\0' };
	int i, j, cnt = 0;

	scanf("%s", str1);
	scanf("%s", str2);

	if (strlen(str1) == strlen(str2))
	{
		for (i = 0; i < strlen(str2); i++)
		{
			if (str1[i] != str2[i])
				break;
		}
		if (i == strlen(str2))
			cnt = 1;
	}
	else
	{
		for (i = 0; i <= strlen(str1) - strlen(str2); i++)
		{
			for (j = 0; j < strlen(str2); j++)
			{
				if (str1[i + j] != str2[j])
					break;
			}
			if (j == strlen(str2))
				cnt = 1;
		}
	}

	printf("%d %d", strlen(str1), cnt);

	return 0;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

int main()
{
	char str1[101] = { '\0' };
	char str2[101] = { '\0' };
	int i, j, cnt = 0;

	gets(str1);
	gets(str2);

	if (strlen(str1) == strlen(str2))
	{
		for (i = 0; i < strlen(str2); i++)
		{
			if (str1[i] != str2[i])
				break;
		}
		if (i == strlen(str2))
			cnt++;
	}
	else
	{
		for (i = 0; i <= strlen(str1) - strlen(str2); i++)
		{
			for (j = 0; j < strlen(str2); j++)
			{
				if (str1[i + j] != str2[j])
					break;
			}

			if (j == strlen(str2))
			{
				cnt++;
				i += j - 1; 
				continue;
			}
		}
	}
	
	printf("%d", cnt);

	return 0;
}
