#pragma warning (disable : 4996)
typedef struct time {
	int x;
	int y;
	int z;
}TIME;

int main()
{	
	TIME a, b;
	int t1, t2;

	scanf("%d %d %d", &a.x, &a.y, &a.z);
	scanf("%d %d %d", &b.x, &b.y, &b.z);

	t1 = a.x * 60 * 60 + a.y * 60 + a.z;
	t2 = b.x * 60 * 60 + b.y * 60 + b.z;

	t1 = t2 - t1;

	a.x = t1 / 3600;
	a.y = t1 % 3600 / 60;
	a.z = t1 % 60;

	printf("%d %d %d", a.x, a.y, a.z);
	
	return 0;
}

#include <stdio.h>
#pragma warning (disable : 4996)
typedef struct stu {
	char name[20];
	int a;
	int b;
	int c;
}STU;

int main()
{	
	STU a[20];
	int i, n;
	double tmp = 0.0;
	char result;

	scanf("%d", &n);

	for (i = 0; i < n; i++)
	{
		scanf("%s %d %d %d", a[i].name, &a[i].a, &a[i].b, &a[i].c);
	}

	for (i = 0; i < n; i++)
	{
		tmp = a[i].a + a[i].b + a[i].c;
		tmp /= 3;

		if (tmp >= 90)
			result = 'A';
		else if (tmp >= 80)
			result = 'B';
		else if (tmp >= 70)
			result = 'C';
		else
			result = 'F';

		printf("%s %.1lf %c\n", a[i].name, tmp, result);
	}

	return 0;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

typedef struct day {
	int year;
	int month;
	int day;
}DAY;

DAY* select_min(DAY *a, DAY *b)
{
	if (a->year < b->year)
		return a;
	else if (a->year > b->year)
		return b;
	else
	{
		if (a->month < b->month)
			return a;
		else if (a->month > b->month)
			return b;
		else
		{
			if (a->day < b->day)
				return a;
			else if (a->day > b->day)
				return b;
			else
				return a;
		}
	}
}

int main()
{
	DAY a[2];
	DAY *p;

	scanf("%d/%d/%d", &a[0].year, &a[0].month, &a[0].day);
	scanf("%d/%d/%d", &a[1].year, &a[1].month, &a[1].day);

	p = select_min(&a[0], &a[1]);

	printf("%d/%d/%d", p->year, p->month, p->day);
	
	return 0;
}

#include <stdio.h>
#include <string.h>
#pragma warning (disable:4996)

typedef struct stu {
	char name[10];
	int a;
	int b;
	int c;
	double avg;
}STU;

void read_data(STU *a)
{
	int i;

	for (i = 0; i < 10; i++)
	{
		scanf("%s %d %d %d", a[i].name, &a[i].a, &a[i].b, &a[i].c);
	}
}

void cal_avg(STU* a)
{
	int i;

	for (i = 0; i < 10; i++)
	{		
		a[i].avg = (a[i].a + a[i].b + a[i].c) / 3.0;
	}
}

void sort(STU* a)
{
	int i, j;
	STU tmp;

	for (i = 0; i <9; i++)
	{
		for (j = 0; j < 9 - i; j++)
		{
			if (a[j].avg < a[j+1].avg )
			{
				tmp = a[j];
				a[j] = a[j + 1];
				a[j + 1] = tmp;
			}
		}
	}
}

void print_score(STU* a)
{
	int i;

	printf("%s %.2lf\n%s %.2lf\n", a[0].name, a[0].avg, a[9].name, a[9].avg);

	for (i = 7; i < 10; i++)
	{
			printf("%s %.2lf\n", a[i].name, a[i].avg);
	}
}

int main()
{
	STU a[10];

	read_data(a);
	cal_avg(a);
	sort(a);
	print_score(a);

	return 0;
}
