		for (j = 0; j < m; j++)
		{
			if (cnt % ('z' - 'a' + 1) == 0)
			{
				if (flag == 1)
					flag = 0;
				else
					flag = 1;
			}

			if (flag == 1)
				ar[i][j] = cnt % ('z' - 'a' + 1) + 'a';

			if (flag == 0)
				ar[i][j] = cnt % ('z' - 'a' + 1) + 'A';

			cnt++;

#include <stdio.h>
#include <stdlib.h>
#pragma warning(disable:4996)

typedef struct stu {
	char name[8];
	int result[3];
	double avg;
}stu;

int main()
{
	int n, i;
	stu *ar;

	scanf("%d", &n);
	
	ar = (stu*)malloc(sizeof(stu)*n);

	for (i = 0; i < n; i++)
	{
		scanf("%s %d %d %d", ar[i].name, &ar[i].result[0], &ar[i].result[1], &ar[i].result[2]);
		ar[i].avg = (ar[i].result[0] + ar[i].result[1] + ar[i].result[2])/3.0;
	}

	for (i = 0; i < n; i++)
	{
		printf("%s %.1lf", ar[i].name, ar[i].avg);

		if (ar[i].result[0] >= 90 || ar[i].result[1] >= 90 || ar[i].result[2] >= 90)
			printf(" GREAT");

		if (ar[i].result[0] < 70 || ar[i].result[1] < 70 || ar[i].result[2] < 70)
			printf(" BAD");
	
		printf("\n");
	}


	free(ar);
	return 0;
}

#include <string.h>
#include <stdlib.h>
#pragma warning(disable:4996)

int main()
{
	int n, i = 0, j, len = 5;
	int *ar = (int*)malloc(sizeof(int) * 5);
	int *tmp;

	while (1)
	{
		scanf("%d", &n);

		if (n == -1 || i > 20)
			break;

		ar[i] = n;
		i++;

		if (i == len)
		{
			len += 3;
			tmp = (int*)malloc(sizeof(int)*len);
			
			for (j = 0; j < len - 3; j++)
				tmp[j] = ar[j];
			
			free(ar);
			ar = tmp;
		}
	}

	for (j = 0; j < i; j++)
		printf(" %d", ar[j]);
	printf(" -1");

	free(ar);
	return 0;
}
