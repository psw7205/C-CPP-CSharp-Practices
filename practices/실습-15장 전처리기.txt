#include <stdio.h>
#pragma warning(disable:4996)

#define ABS(a)		(((a) < 0) ? -(a) : (a))
#define MAX(a, b)	(((a) > (b)) ? (a) : (b))

int main()
{
	int ar[6];
	int max, abs,a,b;

	for (int i = 0; i < 6; i++)
		scanf("%d", &ar[i]);
	
	max = ar[0];
	abs = ABS(max);

	for (int i = 1; i < 6; i++)
	{
		abs = MAX(ABS(max), ABS(ar[i]));
		max = (abs == ABS(max)) ? max : ar[i];
	}

	printf("%d %d", max, abs);

	return 0;
}
