#include <stdio.h>
#pragma warning (disable:4996)

int* smaller(int* ar, int* ar2)
{
	return *ar > *ar2 ? ar2 : ar;
}

int main()
{
	int ar[5] = { 2,1,3,0,4 };
	int* p1;

	p1 = smaller(&ar[1], &ar[3]);

	printf("%d", *p1);

	return 0;
}

#include <stdio.h>
#pragma warning (disable:4996)

void findMinMax(int* ar, int size, int* max, int* min)
{
	int i;
	*max = *min = ar[0];

	for (i = 0; i < size; i++)
	{
		if (*max <= ar[i])
			*max = ar[i];

		if (*min >= ar[i])
			*min = ar[i];
	}
}

int main()
{
	int a[100];
	int i = 0, max, min;


	while (1)
	{
		scanf("%d", &a[i]);

		if (a[i] == -99)
		{
			break;
		}
		i++;
	}

	findMinMax(a, i, &max, &min);
	printf("%d %d", max, min);

	return 0;
}
