문제 1
#include <stdio.h>
#pragma warning(disable:4996)

int main()
{	
	char arr[100];
	int i = 0;
	int big = 0, small = 0, num = 0, etc = 0;
	
	while (1)
	{
		scanf("%c", &arr[i]);
		
		if (arr[i] == '*')
		{
			break;
		}
		else if (arr[i] >= 'A' && arr[i] <= 'Z')
		{
			big++;
		}
		else if (arr[i] >= 'a' && arr[i] <= 'z')
		{
			small++;
		}
		else if (arr[i] >= '0' && arr[i] <= '9')
		{
			num++;
		}
		else
		{
			etc++;
		}
		
		i++;
	}
	
	i--;
	
	for (i; i >= 0; i--)
	{
		printf("%c", arr[i]);
	}
	
	printf("\n%d %d %d %d", big, small, num, etc);
	
	return 0;
}

#include <stdio.h>
#pragma warning(disable:4996)
int main()
{	
	int arr[100] = { 0 };
	int i = 0, j = 0, tmp = 0;
	int n;
	
	scanf("%d", &n);
	
	for (i = 0; i < n; i++)
	{
		scanf("%d", &arr[i]);
	}
	
	for (i = 0; i < n - 1; i++)
	{
		for (j = 0; j < n - 1 - i; j++)
		{
			if (arr[j] < arr[j + 1])
			{
				tmp = arr[j];
				arr[j] = arr[j + 1];
				arr[j + 1] = tmp;
			}
		}
	}
	
	for (i = 0; i < n; i++)
	{
		printf(" %d", arr[i]);
	}
	
	return 0;
}

문제 3
#include <stdio.h>
#pragma warning(disable:4996)

char grade(int a, int b, int c);

int main()
{	
	int a, b, c;
	
	scanf("%d %d %d", &a, &b, &c);
	
	printf("%c", grade(a, b, c));
	
	return 0;
}

char grade(int a, int b, int c)
{	
	double avg;
	
	avg = (double)(a + b + c) / 3;
	
	if (avg >= 91.5)
	{
		return 'A';
	}
	else if (avg >= 82.5)
	{
		return 'B';
	}
	else if (avg >= 72.5)
	{
		return 'C';
	}
	else
	{
		return 'F';
	}
}

문제 4
#include <stdio.h>
#pragma warning(disable:4996)

int main()
{	
	int arr[10] = { 0 };
	int i = 0, n = 0, sum = 0, cnt = 0;
	
	scanf("%d", &n);
	
	while (n != 0)
	{
		arr[i] = n % 10;
		n = n / 10;
		cnt++;
		i++;
	}
	
	for (i = 0; i < cnt; i++)
	{
		sum += arr[i];
	}
	
	printf("%d\n", sum);
	
	for (i = 0; i < cnt / 2; i++)
	{
		if (arr[i] != arr[cnt - i - 1])
		{
			break;
		}
	}
	
	if (i < cnt / 2)
	{
		printf("No");
	}
	else
	{
		printf("Yes");
	}
	
	return 0;
}
