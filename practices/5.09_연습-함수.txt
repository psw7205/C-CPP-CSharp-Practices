#include<stdio.h>
#pragma warning(disable:4996)

int max(int in_a, int in_b)
{
	int c;

	c = (in_a > in_b) ? in_a : in_b;

	return c;
}

int main()
{
	int a, b, c, d;
	int Max;

	scanf("%d %d %d %d", &a, &b, &c, &d);

	Max = max(a, b) > max(c, d) ? max(a, b) : max(c, d);

	printf("%d\n", Max);

	return 0;
}

#include<stdio.h>
#pragma warning(disable:4996)

char atoA(char ch);

int main()
{
	char str[10];
	int i;

	for (i = 0; i < 10; i++)
		scanf("%c", &str[i]);

	for (i = 0; i < 10; i++)
		str[i] = atoA(str[i]);

	for (i = 0; i < 10; i++)
		printf("%c", str[i]);

	return 0;
}

char atoA(char ch)
{
	if (ch >= 'a' && ch <= 'z')
		ch = ch + ('A' - 'a');

	return ch;
}
