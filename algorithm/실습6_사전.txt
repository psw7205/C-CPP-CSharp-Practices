#include <stdio.h>
#include <stdlib.h>
#pragma warning (disable:4996)

int binarySearch(int *ar, int start, int end, int key)
{	int mid = (start + end) / 2;
	if (start > end)
		return end;
	if (ar[mid] < key)
		binarySearch(ar, mid + 1, end, key);
	else if (ar[mid] > key)
		binarySearch(ar, start, mid - 1, key);
	else
		return mid;
}

int main()
{	
	int n, i, key, tmp;
	int *ar;
	scanf("%d %d", &n, &key);
	ar = (int*)malloc(sizeof(int)*n);
	
	for (i = 0; i < n; i++)
	{
		scanf("%d", &ar[i]);
	}
	
	tmp = binarySearch(ar, 0, n-1, key);
	printf("%d", tmp);
	
	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#pragma warning (disable:4996)

int binarySearch(int *ar, int start, int end, int key)
{
	int mid = (start + end) / 2;

	while (start <= end)
	{
		if (ar[mid] < key)
		{
			start = mid + 1;
			mid = (mid + 1 + end) / 2;
		}
		else if (ar[mid] > key)
		{
			end = mid - 1;
			mid = (start + mid - 1) / 2;
		}
		else
			return mid;
	}

	return start;
}

int main()
{
	int n, i, key, tmp;
	int *ar;

	scanf("%d %d", &n, &key);

	ar = (int*)malloc(sizeof(int)*n);

	for (i = 0; i < n; i++)
	{
		scanf("%d", &ar[i]);
	}

	tmp = binarySearch(ar, 0, n-1, key);

	printf("%d", tmp);

	return 0;
}

#include <stdio.h>

#pragma warning (disable:4996)

int search(int a, int b, int n, char* str)
{
	int m, i;
	for (i = 0; i < n; i++)
	{
		m = (a + b) / 2;
		
		if (str[i] == 'Y')
		{
			a = m + 1;
		}

		if (str[i] == 'N')
		{
			b = m;
		}
	}

	return a;
}
